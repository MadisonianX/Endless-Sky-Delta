mission "Sol Passenger"
	name "Passenger transport to <planet>"
	job
	repeat
	description "A passenger needs a quick lift to <destination>. Payment is <payment>. Passenger License required."
	passengers 1
	to offer
		random < 80
	to accept
		has "license: Passenger"
	source
		government "Republic (Sol)"
	destination
		government "Republic (Sol)"
	on visit
		dialog phrase "generic passenger on visit"
	on complete
		payment
		payment 300 
		dialog phrase "generic passenger dropoff payment"

# --------- SOL CARGO ---------

mission "Intro Cargo [Intrasolar 1]"
	job
	name "Water Coolant System to <planet>"
	description "Buy a Water Coolant System from the outfitter and deliver it to <planet>. Payment is <payment>."
	source 
		government "Republic (Sol)"
		outfits "Water Coolant System"
	destination "Icarus Station"
	to offer
		random < 30
		not "Intro Cargo [Intrasolar 1]: done"
	on accept
		conversation
			scene "outfit/water cooling"
			scene "scene/outfit to cargo"
			`	Sometimes you may want to buy some outfits directly into your cargo hold instead of having them installed. To accomplish this, you need to switch out of installation mode into 'purchase to cargo' mode. This can be achieved by either checking the "show outfits in cargo" box at the bottom left, or Control+clicking on your selected ship to deselect it.`
			`	When you are in 'purchase to cargo' mode, your ship information on the right hand side will be replaced by information on how much cargo space you currently have available.`
			`	Outfits that are in your cargo hold can be (re)installed at any outfitter without cost.`
	on complete
		outfit "Water Coolant System" -1
		payment 18400
		dialog `You deliver the Water Coolant System to an ice cream shop. The owner pays you <payment> and gives you a free scoop of your favorite flavor.`

	deadline 7
	name "Mining: Lead to <planet>"

		government "Republic (Sol)"
	to offer
		random < 30
		not "Intro Sol Mining [1]: done"

		payment 4500
		dialog `The municipal authorities on <planet> accept your delivery of Lead and pay you <payment>.`
